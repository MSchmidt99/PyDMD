true_Z = {
    'Z': 0.95 + 0.2j,
    'Xi': 0+1j,
}
s = np.zeros(200, dtype=np.complex128)
s[10:110] += (true_Z['Z'] ** np.arange(100)) * true_Z['Xi']
s_centered = s.real
s_uncentered = s_centered - np.mean(s_centered)
plt.plot(s_uncentered)
